## CommandGeneration 模块

### 位置
`src/core/command-generation/`

### 职责
生成 AI 工具的指令文件（commands 和 skills）。

### 核心类：CommandGenerator

```typescript
// src/core/command-generation/generator.ts

export class CommandGenerator {
  private adapters: Map<string, ToolCommandAdapter>;

  constructor() {
    this.adapters = new Map();
    // 注册所有适配器
    for (const adapter of allAdapters) {
      this.adapters.set(adapter.toolId, adapter);
    }
  }

  // 为单个工具生成命令
  generateForTool(toolId: string, commands: CommandContent[]): GeneratedCommand[] {
    const adapter = this.adapters.get(toolId);
    if (!adapter) throw new Error(`Unknown tool: ${toolId}`);

    return commands.map(cmd => ({
      path: adapter.getFilePath(cmd.id),
      fileContent: adapter.formatFile(cmd),
    }));
  }

  // 为多个工具生成
  generateForTools(toolIds: string[], commands: CommandContent[]): GeneratedCommand[] {
    return toolIds.flatMap(id => this.generateForTool(id, commands));
  }
}
```

## CompletionProvider 模块

### 位置
`src/core/completions/`

### 职责
为各种 Shell 生成自动补全脚本。

### 核心类

```typescript
// src/core/completions/completion-provider.ts

export class CompletionProvider {
  private registry: CommandRegistry;

  constructor() {
    this.registry = new CommandRegistry();
    this.registerCommands();
  }

  // 获取补全数据
  getCompletions(type: string): string[] {
    switch (type) {
      case 'commands':
        return this.registry.getCommandNames();
      case 'changes':
        return this.getChangeNames();
      case 'schemas':
        return this.getSchemaNames();
      default:
        return [];
    }
  }

  private getChangeNames(): string[] {
    // 调用 openspec list --json 获取
    const result = execSync('openspec list --json');
    const data = JSON.parse(result.toString());
    return data.changes.map((c: any) => c.name);
  }
}
```

### Shell Generator

每种 Shell 有对应的 Generator：

```typescript
// src/core/completions/generators/bash-generator.ts

export class BashGenerator implements ShellGenerator {
  generate(provider: CompletionProvider): string {
    const commands = provider.getCompletions('commands');

    return `
# OpenSpec Bash Completion
_openspec_completions() {
  local cur="\${COMP_WORDS[COMP_CWORD]}"
  local commands="${commands.join(' ')}"

  if [[ \${COMP_CWORD} -eq 1 ]]; then
    COMPREPLY=($(compgen -W "\${commands}" -- "\${cur}"))
  fi
}

complete -F _openspec_completions openspec
`;
  }
}
```

## Parser 模块

### 位置
`src/core/parsers/`

### 职责
解析 Markdown 和变更文件。

### ChangeParser

```typescript
// src/core/parsers/change-parser.ts

export class ChangeParser {
  // 解析 proposal.md
  parseProposal(content: string): Proposal {
    return {
      why: this.extractSection(content, 'Why'),
      whatChanges: this.extractSection(content, 'What Changes'),
      capabilities: this.parseCapabilities(content),
      impact: this.extractSection(content, 'Impact'),
    };
  }

  // 解析 tasks.md
  parseTasks(content: string): Task[] {
    const tasks: Task[] = [];
    const lines = content.split('\n');

    for (const line of lines) {
      const match = line.match(/^- \[([ x])\] (\d+\.\d+) (.+)$/);
      if (match) {
        tasks.push({
          completed: match[1] === 'x',
          id: match[2],
          description: match[3],
        });
      }
    }

    return tasks;
  }
}
```

### MarkdownParser

```typescript
// src/core/parsers/markdown-parser.ts

export class MarkdownParser {
  // 提取所有 Requirements
  extractRequirements(content: string): Requirement[] {
    const requirements: Requirement[] = [];
    const regex = /### Requirement: (.+?)\n([\s\S]*?)(?=### |$)/g;

    let match;
    while ((match = regex.exec(content)) !== null) {
      requirements.push({
        name: match[1],
        content: match[2].trim(),
        scenarios: this.extractScenarios(match[2]),
      });
    }

    return requirements;
  }

  // 提取所有 Scenarios
  private extractScenarios(content: string): Scenario[] {
    const scenarios: Scenario[] = [];
    const regex = /#### Scenario: (.+?)\n([\s\S]*?)(?=#### |### |$)/g;

    let match;
    while ((match = regex.exec(content)) !== null) {
      scenarios.push({
        name: match[1],
        steps: this.parseSteps(match[2]),
      });
    }

    return scenarios;
  }
}
```
