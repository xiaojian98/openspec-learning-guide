# 19 - 验证系统

> 了解 OpenSpec 如何使用 Zod 进行数据验证

## Zod 简介

[Zod](https://zod.dev/) 是一个 TypeScript 优先的 schema 声明和验证库。

### 基本用法

```typescript
import { z } from 'zod';

// 定义 Schema
const UserSchema = z.object({
  name: z.string(),
  age: z.number().int().positive(),
  email: z.string().email().optional(),
});

// 推断类型
type User = z.infer<typeof UserSchema>;

// 验证
const user = UserSchema.parse({
  name: 'Alice',
  age: 30,
}); // ✅ 通过

const invalid = UserSchema.parse({
  name: 'Bob',
  age: -1,
}); // ❌ 抛出 ZodError
```

## 配置验证

### 全局配置

```typescript
// src/core/global-config.ts

export type Profile = 'core' | 'custom';
export type Delivery = 'both' | 'skills' | 'commands';

export interface GlobalConfig {
  featureFlags?: Record<string, boolean>;
  profile?: Profile;
  delivery?: Delivery;
  workflows?: string[];
}

const DEFAULT_CONFIG: GlobalConfig = {
  featureFlags: {},
  profile: 'core',
  delivery: 'both',
};

export function getGlobalConfig(): GlobalConfig {
  const configPath = getGlobalConfigPath();

  try {
    if (!fs.existsSync(configPath)) {
      return { ...DEFAULT_CONFIG };
    }

    const content = fs.readFileSync(configPath, 'utf-8');
    const parsed = JSON.parse(content);

    // 合并默认值
    const merged: GlobalConfig = {
      ...DEFAULT_CONFIG,
      ...parsed,
      featureFlags: {
        ...DEFAULT_CONFIG.featureFlags,
        ...(parsed.featureFlags || {}),
      },
    };

    return merged;
  } catch (error) {
    // 解析失败返回默认值
    return { ...DEFAULT_CONFIG };
  }
}
```

## 验证流程

### validate 命令

```typescript
// src/commands/validate.ts

export class ValidateCommand {
  async execute(itemName?: string, options?: ValidateOptions): Promise<void> {
    const items = this.getItemsToValidate(itemName, options);

    for (const item of items) {
      const result = await this.validateItem(item);

      if (!result.valid) {
        this.reportErrors(item, result.errors);
      }
    }
  }

  private async validateItem(item: string): Promise<ValidationResult> {
    // 1. 解析文件
    const content = await fs.readFile(item, 'utf-8');

    // 2. 根据类型选择 Schema
    const schema = this.getSchemaForItem(item);

    // 3. 验证
    const result = schema.safeParse(content);

    if (result.success) {
      return { valid: true };
    } else {
      return {
        valid: false,
        errors: result.error.issues.map(this.formatError),
      };
    }
  }
}
```

## 错误格式化

### ZodError 处理

```typescript
function formatZodError(error: z.ZodError): string[] {
  return error.issues.map((issue) => {
    const path = issue.path.join('.');
    return `${path}: ${issue.message}`;
  });
}

// 示例输出：
// [
//   'artifacts.0.id: Artifact ID is required',
//   'artifacts.1.requires: Invalid type',
// ]
```

### 用户友好的错误消息

```typescript
function formatErrorForUser(issue: z.ZodIssue): string {
  const path = issue.path.length > 0 ? issue.path.join('.') : 'root';

  switch (issue.code) {
    case 'invalid_type':
      return `Expected ${issue.expected} at ${path}, got ${issue.received}`;
    case 'too_small':
      return `${path} is too short (minimum ${issue.minimum})`;
    case 'too_big':
      return `${path} is too long (maximum ${issue.maximum})`;
    default:
      return `${path}: ${issue.message}`;
  }
}
```
